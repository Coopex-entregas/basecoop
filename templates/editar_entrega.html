from flask import request, redirect, url_for, flash, render_template, session from models import db, Entrega, Cooperado @app.route('/editar_entrega/', methods=['GET', 'POST']) def editar_entrega(entrega_id): entrega = Entrega.query.get_or_404(entrega_id) user_tipo = session.get('usuario_tipo', 'cooperado') # Ajuste conforme sua sessão if request.method == 'POST': # Pega dados do formulário (se enviados) descricao = request.form.get('descricao') valor = request.form.get('valor') cooperado_id = request.form.get('cooperado_id') status_pagamento = request.form.get('status_pagamento') status_entrega = request.form.get('status_entrega') # Só altera se for admin if user_tipo == 'adm': if descricao is not None: entrega.descricao = descricao if valor is not None: try: entrega.valor = float(valor) except ValueError: flash("Valor inválido.", "error") return redirect(url_for('editar_entrega', entrega_id=entrega_id)) if cooperado_id: if cooperado_id.isdigit(): entrega.cooperado_id = int(cooperado_id) from datetime import datetime entrega.hora_atribuida = datetime.utcnow() else: entrega.cooperado_id = None entrega.hora_atribuida = None # Atualiza status de pagamento se enviado if status_pagamento in ['pendente', 'pago']: entrega.status_pagamento = status_pagamento # Atualiza status de entrega se enviado if status_entrega in ['pendente', 'em rota', 'entregue']: entrega.status_entrega = status_entrega db.session.commit() flash("Entrega atualizada com sucesso.", "success") # Redireciona para dashboard adequado, por exemplo: if user_tipo == 'adm': return redirect(url_for('dashboard')) else: return redirect(url_for('dashboard_cooperado')) # GET request - renderiza página de edição cooperados = Cooperado.query.order_by(Cooperado.nome).all() if user_tipo == 'adm' else [] return render_template('editar_entrega.html', entrega=entrega, cooperados=cooperados, user_tipo=user_tipo)
